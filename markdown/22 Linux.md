```shell
su root	#切换到root用户 vagrant密码也是vagrant
cd ..	# 返回上级目录	change directory
pwd		# 显示当前目录	print working directory 
ls		# 				list
touch 	# 新建文件
rm		# 删除文件		remove  
rm -r	# 删除文件夹
mkdir	# 新建路径		make directory
mv a b	# 把a移到b			move
clear	# 清屏
history	# 查看历史命令
help	# 帮助
exit	# 退出
```

```cmd
vagrant up		# 启动虚拟机
ctrl + C 		# 打断程序
vagrant ssh		# 连接虚拟机
# 虚拟机密码是vagrant
```
